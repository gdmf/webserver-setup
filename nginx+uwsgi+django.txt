# src: http://www.collabspot.com/2012/08/14/setting-up-nginx-uwsgi-python-ubuntu-12-04/
# installing nginx
$ /etc/init.d/nginx start

# installing uWSGI
$ apt-get install uwsgi
# or
$ pip install uwsgi

# folder structure
$ mkdir ~/projects/project1
($ sudo chown <user>:<user> project1)
$ cd ~/projects/project1
$ mkdir conf src logs

~/projects/
`- project1/
   |- venv/
   |- conf/
   |  |- nginx.conf (include in main nginx.conf)
   |  `- uwsgi.ini 
   |- src/
   |  `- example/ (django-admin.py startproject example)
   `- logs/
      |- access.log (nginx access log)
      `- error.log (nginx error log)

# venv/  -- this is where the virtual environment lives
$ virtualenv ~/projects/project1/venv
$ source ~/projects/project1/venv/bin/activate
(venv)$ pip install django

# src/  -- this is where the project's source code lives
$ cd ~/projects/project1/src
(venv)$ django-admin.py startproject example

# conf/  -- this is where config files for Nginx and uWSGI are stored
# conf/uwsgi.ini

[uwsgi]
# variables
projectname = example
projectdomain = example.com
base = /var/www/example.com
# config
protocol = uwsgi
venv = %(base)/venv
pythonpath = %(base)/src/%(projectname)
module = %(projectname).wsgi
socket = /tmp/%(projectdomain).sock
logto = %(base)/logs/uwsgi.log
